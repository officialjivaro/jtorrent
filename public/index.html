Frontend example

const manifest = await fetch('https://officialjivaro.github.io/jtorrent/data/manifest.json').then(r => r.json());
const shardUrls = manifest.sharding.search_shards.map(s => 'https://officialjivaro.github.io/jtorrent/' + s.path);
const firstSearchShard = await fetch(shardUrls[0]).then(r => r.json());
